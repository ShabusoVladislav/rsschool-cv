function indexEqualsValue(a) {
    let left = 0;
    let right = a.length - 1;
    let mid;
    let min;
    while (left <= right) {
        mid = Math.round((right - left) / 2 + left);
        if (mid === a[mid]) {
            min = mid;
            right = mid - 1;
        }
        else if (a[mid] > mid) right = mid - 1;
        else left = mid + 1;
    }
    return min !== undefined ? min : -1;
}
